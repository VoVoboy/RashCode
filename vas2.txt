

public class Ret {
	
		int fact(int n) {
			int result;
			
			if(n==1) return 1;
			result = fact(n-1)*n;
			return result;
			
		}
	
}
public class Op {
	public static void main(String args[]){
		
		Ret f = new Ret();
		
		System.out.println("Факториал 3 равен: " + f.fact(3));
		System.out.println("Факторифл 4 равен: " + f.fact(4));
		System.out.println("Факториал 5 равен: " + f.fact(5));
	}

}
